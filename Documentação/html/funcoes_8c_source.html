96 void linha(pixel **p, reta r) // Função de desenhos de linhas
97 {
98  int dy = abs(r.p2.y - r.p1.y); //Diferença entre os pontos y2 e y1
99  int dx = abs(r.p2.x - r.p1.x); //Diferença entre os pontos x2 e x1
100  int sx, sy; //Variaveis que serão utilizadas para armazenar o sinal de x e y
101  int i = r.p1.y;
102  int j = r.p1.x;
103  if (r.p1.x < r.p2.x) // Determinar os sinais de x e y
104  {
105  sx = 1;
106  }
107  else
108  {
109  sx = -1;
110  }
111  if (r.p1.y < r.p2.y)
112  {
113  sy = 1;
114  }
115  else
116  {
117  sy = -1;
118  } //fim da determinação de sinais
119  int erro;
120  int erroTemp;
121  if (dx > dy) // Definição do erro inicial baseado na maior diferença
122  {
123  erro = dx / 2;
124  }
125  else
126  {
127  erro = (-dy) / 2;
128  }
129  while (true)
130  {
131  p[i][j] = color(p[i][j], cor.red, cor.green, cor.blue);
132  if ((i == r.p2.y) && (j == r.p2.x)) // Testa se chegou ao ponto final da reta
133  {
134  break;
135  }
136  erroTemp = erro;
137  if (erroTemp > -dx)
138  {
139  erro -= dy;
140  j += sx;
